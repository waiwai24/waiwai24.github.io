性能监控包括响应时间和 TPS(每秒事务处理量)测试:

```
export async function testPerformance(url: string, model: string) {
  const startTime = performance.now();
  
  try {
    const response = await fetch(`${url}/api/generate`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: model,
        prompt: "Hello", // 使用简单prompt测试性能
        stream: false
      })
    });
    
    const endTime = performance.now();
    const responseTime = endTime - startTime;

    return {
      status: 'success',
      responseTime,
      success: response.ok
    };
  } catch (error) {
    return {
      status: 'error',
      message: error.message
    };
  }
}

// TPS测试
export async function testTPS(url: string, model: string, duration: number = 10000) {
  const requests = [];
  const startTime = performance.now();
  let completedRequests = 0;
  
  while (performance.now() - startTime < duration) {
    requests.push(testPerformance(url, model));
    completedRequests++;
  }
  
  await Promise.all(requests);
  
  const totalTime = (performance.now() - startTime) / 1000; // 转换为秒
  const tps = completedRequests / totalTime;
  
  return {
    tps: Math.round(tps),
    totalRequests: completedRequests,
    duration: totalTime
  };
}
```
